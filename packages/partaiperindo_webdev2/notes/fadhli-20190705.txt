SELECT
RIGHT(users.tanggal_lahir, 4) AS `Tahun`,
MID(users.tanggal_lahir, 2, 2) AS `Bulan`,
LEFT(users.tanggal_lahir, 2) AS `Tanggal`
FROM
users;



UPDATE
users
SET
users.tanggal_lahir2 = CONCAT(
	RIGHT(users.tanggal_lahir, 4),
	'-',
	MID(users.tanggal_lahir, 4, 2),
	'-',
	LEFT(users.tanggal_lahir, 2)
)
WHERE
MID(users.tanggal_lahir, 4, 2) <= 12;
